51 def _classify_file(self, filepath: Path, content: str) -> str:
52 """Classify file type based on path and content."""
53 path_parts = filepath.parts
54
55 if "cfg" in path_parts:
56 return "configuration"
57 elif "frontmatter" in path_parts:
58 return "frontmatter"
59 elif "chapters" in path_parts:
60 if any(
61 template in content for template in ["JournalArticle", "addarticle"]
62 ):
63 return "article"
64 else:
65 return "chapter"
66 elif "articles" in path_parts:
67 return "full_article"
68 elif filepath.name == "main.tex":
69 return "main_document"
70 elif filepath.name == "preamble.tex":
71 return "preamble"
72 else:
73 return "other"

75 def _extract_sections(self, content: str) -> List[Dict]:
76 """Extract section hierarchy."""
77 sections = []
78 section_pattern = (
79 r"\\(section|subsection|subsubsection|paragraph)\*?\{([^}]+)\}"
80 )
81
82 for match in re.finditer(section_pattern, content):
83 sections.append(
84 {
85 "level": match.group(1),
86 "title": match.group(2).strip(),
87 "type": "structure",
88 }
89 )
90
91 return sections

93 def _extract_math(self, content: str) -> List[Dict]:
94 """Extract mathematical content."""
95 math_content = []
96
97 # Display math
98 display_pattern = r"\\\‍[([^\\]+)\\\‍]"
99 for match in re.finditer(display_pattern, content):
100 math_content.append(
101 {
102 "type": "display_math",
103 "content": match.group(1).strip(),
104 "format": "tex",
105 }
106 )
107
108 # Inline math
109 inline_pattern = r"\$([^$]+)\$"
110 for match in re.finditer(inline_pattern, content):
111 math_content.append(
112 {
113 "type": "inline_math",
114 "content": match.group(1).strip(),
115 "format": "tex",
116 }
117 )
118
119 # Equation environments
120 equation_pattern = r"\\begin\{(equation|align|gather)\}([^\\]+)\\end\{\1\}"
121 for match in re.finditer(equation_pattern, content):
122 math_content.append(
123 {
124 "type": "equation_environment",
125 "env": match.group(1),
126 "content": match.group(2).strip(),
127 "format": "tex",
128 }
129 )
130
131 return math_content

133 def _extract_citations(self, content: str) -> List[str]:
134 """Extract bibliographic citations."""
135 citations = []
136
137 # Biblatex citations
138 citation_pattern = r"\\(cite|citep|citet|parencite)\{([^}]+)\}"
139 for match in re.finditer(citation_pattern, content):
140 citations.append(match.group(2))
141
142 return list(set(citations))

144 def _extract_tables(self, content: str) -> List[Dict]:
145 """Extract table structures."""
146 tables = []
147
148 table_pattern = r"\\begin\{tabular\}.*?\\end\{tabular\}"
149 for match in re.finditer(table_pattern, content, re.DOTALL):
150 tables.append({"type": "table", "content": match.group(0), "format": "tex"})
151
152 return tables

154 def _extract_figures(self, content: str) -> List[Dict]:
155 """Extract figure references."""
156 figures = []
157
158 figure_pattern = r"\\begin\{figure\}.*?\\end\{figure\}"
159 for match in re.finditer(figure_pattern, content, re.DOTALL):
160 figures.append(
161 {"type": "figure", "content": match.group(0), "format": "tex"}
162 )
163
164 # Includegraphics
165 include_pattern = r"\\includegraphics[^{]*\{([^}]+)\}"
166 for match in re.finditer(include_pattern, content):
167 figures.append({"type": "image", "file": match.group(1), "format": "tex"})
168
169 return figures

171 def _detect_languages(self, content: str) -> List[str]:
172 """Detect content languages."""
173 languages = []
174
175 # Check for Russian text
176 if re.search(r"[а-яА-Я]", content):
177 languages.append("russian")
178
179 # Check for English text
180 if re.search(r"[a-zA-Z]{3,}", content) and "begin{english}" in content:
181 languages.append("english")
182
183 # Check for other language switches
184 if "begin{russian}" in content:
185 languages.append("russian")
186 if "begin{english}" in content:
187 languages.append("english")
188
189 return list(set(languages))

191 def _analyze_structure(self, content: str) -> Dict:
192 """Analyze document structure."""
193 structure = {
194 "has_abstract": bool(
195 re.search(r"\\(RUAbstract|ENAbstract|abstract)", content)
196 ),
197 "has_keywords": bool(re.search(r"(Keywords|Ключевые слова)", content)),
198 "has_bibliography": bool(
199 re.search(r"\\(bibliography|printbibliography)", content)
200 ),
201 "has_toc": bool(re.search(r"\\tableofcontents|\\printtoc", content)),
202 "article_count": len(re.findall(r"\\(addarticle|JournalArticle)", content)),
203 "page_count_estimate": len(content) // 2000, # Rough estimate
204 }
205
206 return structure

208 def reconstruct_markdown_tex(self, analysis: Dict) -> str:
209 """Reconstruct markdown+TeX from analysis."""
210 if analysis.get("error"):
211 return f"# Error\n\n{analysis['error']}"
212
213 file_type = analysis["type"]
214
215 if file_type == "configuration":
216 return self._reconstruct_config(analysis)
217 elif file_type in ["article", "full_article"]:
218 return self._reconstruct_article(analysis)
219 elif file_type == "frontmatter":
220 return self._reconstruct_frontmatter(analysis)
221 elif file_type == "main_document":
222 return self._reconstruct_main(analysis)
223 else:
224 return self._reconstruct_generic(analysis)

226 def _reconstruct_article(self, analysis: Dict) -> str:
227 """Reconstruct article content as markdown+TeX."""
228 md_content = []
229
230 # Extract title from first section or metadata
231 sections = analysis["sections"]
232 if sections:
233 title = sections[0]["title"]
234 md_content.append(f"# {title}")
235 md_content.append("")
236
237 # Add abstract if present
238 if analysis["structure"]["has_abstract"]:
239 # This would need more sophisticated parsing
240 md_content.append("## Abstract")
241 md_content.append("")
242 md_content.append("[Abstract content to be extracted from LaTeX]")
243 md_content.append("")
244
245 # Process sections in order
246 current_level = 1
247 for section in analysis["sections"]:
248 level_map = {
249 "section": 1,
250 "subsection": 2,
251 "subsubsection": 3,
252 "paragraph": 4,
253 }
254
255 level = level_map.get(section["level"], 2)
256 md_content.append(f"{'#' * level} {section['title']}")
257 md_content.append("")
258
259 # Add mathematical content
260 if analysis["math_content"]:
261 md_content.append("## Mathematical Content")
262 md_content.append("")
263 for math_item in analysis["math_content"]:
264 if math_item["type"] == "display_math":
265 md_content.append(f"$$\n{math_item['content']}\n$$")
266 md_content.append("")
267 elif math_item["type"] == "inline_math":
268 md_content.append(f"Inline math: ${math_item['content']}$")
269 md_content.append("")
270
271 # Add citations
272 if analysis["citations"]:
273 md_content.append("## References")
274 md_content.append("")
275 for citation in analysis["citations"]:
276 md_content.append(f"- [{citation}]")
277 md_content.append("")
278
279 return "\n".join(md_content)

320def main():

322 analyzer = TeXAnalyzer()

324 # Find all .tex files

344 "languages_found": set(),

346 "total_articles": 0,

348

353 summary["languages_found"].update(analysis.get("languages", []))
354 summary["total_math_content"] += len(analysis.get("math_content", []))
355 summary["total_articles"] += analysis.get("structure", {}).get(

367 json.dump(summary, f, indent=2, ensure_ascii=False)

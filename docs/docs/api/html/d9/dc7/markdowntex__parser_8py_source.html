77 def __init__(self):
78 # Regex patterns for parsing
79 self.patterns = {
80 "heading": re.compile(r"^(#{1,6})\s+(.+)$", re.MULTILINE),
81 "math_block": re.compile(r"\$\$(.*?)\$\$", re.DOTALL),
82 "inline_math": re.compile(r"\$([^$]+)\$"),
83 "code_block": re.compile(
84 r"^```(\w+)?\s*\n(.*?)\n```", re.MULTILINE | re.DOTALL
85 ),
86 "inline_code": re.compile(r"`([^`]+)`"),
87 "link": re.compile(r"\‍[([^\‍]]+)\‍]\‍(([^)]+)\‍)"),
88 "image": re.compile(r"!\‍[([^\‍]]*)\‍]\‍(([^)]+)\‍)"),
89 "emphasis": re.compile(r"\*([^*]+)\*"),
90 "strong": re.compile(r"\*\*([^*]+)\*\*"),
91 "list_item": re.compile(r"^\s*[-*+]\s+(.+)$", re.MULTILINE),
92 "ordered_list_item": re.compile(r"^\s*\d+\.\s+(.+)$", re.MULTILINE),
93 "quote": re.compile(r"^>\s+(.+)$", re.MULTILINE),
94 "table_row": re.compile(r"^\|(.+)\|$", re.MULTILINE),
95 "metadata": re.compile(
96 r"^---\s*\n(.*?)\n---\s*\n", re.MULTILINE | re.DOTALL
97 ),
98 }
99
100 # TeX block patterns
101 self.tex_patterns = {
102 "equation": re.compile(
103 r"\\begin\{equation\}(.*?)\\end\{equation\}", re.DOTALL
104 ),
105 "align": re.compile(
106 r"\\begin\{align\*?\}(.*?)\\end\{align\*?\}", re.DOTALL
107 ),
108 "gather": re.compile(r"\\begin\{gather\}(.*?)\\end\{gather\}", re.DOTALL),
109 "theorem": re.compile(
110 r"\\begin\{theorem\}(.*?)\\end\{theorem\}", re.DOTALL
111 ),
112 "proof": re.compile(r"\\begin\{proof\}(.*?)\\end\{proof\}", re.DOTALL),

115 def parse(self, markdown_text: str) -> ASTNode:
116 """Parse MarkdownTeX text into AST"""
117 # Extract metadata first
118 metadata = self._extract_metadata(markdown_text)
119
120 # Remove metadata from content
121 if metadata:
122 content_text = self.patterns["metadata"].sub("", markdown_text).strip()
123 else:
124 content_text = markdown_text
125
126 # Create document node
127 doc_node = ASTNode(
128 type=NodeType.DOCUMENT,
129 content=[],
130 attributes={"metadata": metadata} if metadata else {},
131 )
132
133 # Parse content blocks
134 blocks = self._split_into_blocks(content_text)
135
136 for block in blocks:
137 parsed_block = self._parse_block(block)
138 if parsed_block:
139 if isinstance(parsed_block, list):
140 doc_node.content.extend(parsed_block)
141 else:
142 doc_node.content.append(parsed_block)
143
144 return doc_node

146 def _extract_metadata(self, text: str) -> Optional[Dict[str, Any]]:
147 """Extract YAML-style metadata from the beginning of the text"""
148 match = self.patterns["metadata"].match(text)
149 if match:
150 try:
151 import yaml
152
153 return yaml.safe_load(match.group(1))
154 except ImportError:
155 # Fallback to simple key-value parsing
156 metadata = {}
157 lines = match.group(1).strip().split("\n")
158 for line in lines:
159 if ":" in line:
160 key, value = line.split(":", 1)
161 metadata[key.strip()] = value.strip()
162 return metadata
163 return None

165 def _split_into_blocks(self, text: str) -> List[str]:
166 """Split text into logical blocks"""
167 # Handle different block types
168 blocks = []
169 current_block = ""
170 in_code_block = False
171 in_math_block = False
172 code_fence = None
173
174 lines = text.split("\n")
175
176 for line in lines:
177 # Check for code fence
178 if line.strip().startswith("```"):
179 if not in_code_block:
180 if current_block.strip():
181 blocks.append(current_block.strip())
182 current_block = ""
183 in_code_block = True
184 code_fence = line.strip()
185 current_block = line + "\n"
186 else:
187 current_block += line + "\n"
188 blocks.append(current_block.strip())
189 current_block = ""
190 in_code_block = False
191 code_fence = None
192 continue
193
194 # Check for math block
195 if "$$" in line and not in_math_block:
196 if current_block.strip():
197 blocks.append(current_block.strip())
198 current_block = ""
199 in_math_block = True
200 current_block = line + "\n"
201 continue
202 elif "$$" in line and in_math_block:
203 current_block += line + "\n"
204 blocks.append(current_block.strip())
205 current_block = ""
206 in_math_block = False
207 continue
208
209 current_block += line + "\n"
210
211 # Block separator for non-code/math content
212 if not in_code_block and not in_math_block and line.strip() == "":
213 if current_block.strip():
214 blocks.append(current_block.strip())
215 current_block = ""
216
217 # Add remaining content
218 if current_block.strip():
219 blocks.append(current_block.strip())
220
221 return [b for b in blocks if b.strip()]

223 def _parse_block(self, block: str) -> Union[ASTNode, List[ASTNode], None]:
224 """Parse a single block of content"""
225 if not block.strip():
226 return None
227
228 # Math block
229 if "$$" in block:
230 return self._parse_math_block(block)
231
232 # Code block
233 if block.strip().startswith("```"):
234 return self._parse_code_block(block)
235
236 # TeX blocks
237 for tex_type, pattern in self.tex_patterns.items():
238 if pattern.match(block):
239 return self._parse_tex_block(block, tex_type)
240
241 # Heading
242 heading_match = self.patterns["heading"].match(block.strip())
243 if heading_match:
244 level = len(heading_match.group(1))
245 text = heading_match.group(2)
246 content = self._parse_inline(text)
247 return ASTNode(
248 type=NodeType.HEADING, content=content, attributes={"level": level}
249 )
250
251 # List
252 if self.patterns["list_item"].match(block) or self.patterns[
253 "ordered_list_item"
254 ].match(block):
255 return self._parse_list(block)
256
257 # Table
258 if "|" in block and block.strip().startswith("|"):
259 return self._parse_table(block)
260
261 # Quote
262 if block.strip().startswith(">"):
263 return self._parse_quote(block)
264
265 # Thematic break
266 if block.strip().replace("-", "").strip() == "":
267 return ASTNode(type=NodeType.THEMATIC_BREAK)
268
269 # Paragraph (default)
270 return self._parse_paragraph(block)

301 def _parse_list(self, text: str) -> ASTNode:
302 """Parse list block"""
303 lines = text.strip().split("\n")
304 items = []
305 is_ordered = False
306
307 for line in lines:
308 ordered_match = self.patterns["ordered_list_item"].match(line)
309 unordered_match = self.patterns["list_item"].match(line)
310
311 if ordered_match:
312 if not is_ordered:
313 is_ordered = True
314 item_content = self._parse_inline(ordered_match.group(1))
315 items.append(ASTNode(type=NodeType.LIST_ITEM, content=item_content))
316 elif unordered_match:
317 item_content = self._parse_inline(unordered_match.group(1))
318 items.append(ASTNode(type=NodeType.LIST_ITEM, content=item_content))
319
320 return ASTNode(
321 type=NodeType.LIST, content=items, attributes={"ordered": is_ordered}

324 def _parse_table(self, text: str) -> ASTNode:
325 """Parse table block"""
326 lines = [
327 line for line in text.strip().split("\n") if line.strip().startswith("|")
328 ]
329 rows = []
330
331 for line in lines:
332 cells = [cell.strip() for cell in line.split("|")[1:-1]]
333 cell_nodes = []
334
335 for cell in cells:
336 cell_content = self._parse_inline(cell)
337 cell_nodes.append(
338 ASTNode(type=NodeType.TABLE_CELL, content=cell_content)
339 )
340
341 rows.append(ASTNode(type=NodeType.TABLE_ROW, content=cell_nodes))
342
343 return ASTNode(type=NodeType.TABLE, content=rows)

345 def _parse_quote(self, text: str) -> ASTNode:
346 """Parse blockquote"""
347 lines = text.strip().split("\n")
348 quoted_lines = []
349
350 for line in lines:
351 if line.strip().startswith(">"):
352 quoted_text = line.strip()[1:].strip()
353 quoted_lines.append(quoted_text)
354
355 quote_content = "\n".join(quoted_lines)
356 parsed_content = self._parse_inline(quote_content)
357
358 return ASTNode(type=NodeType.QUOTE, content=parsed_content)

365 def _parse_inline(self, text: str) -> List[ASTNode]:
366 """Parse inline content"""
367 nodes = []
368 remaining_text = text
369
370 # Process patterns in order of specificity
371 patterns = [
372 ("image", self.patterns["image"]),
373 ("link", self.patterns["link"]),
374 ("inline_code", self.patterns["inline_code"]),
375 ("inline_math", self.patterns["inline_math"]),
376 ("strong", self.patterns["strong"]),
377 ("emphasis", self.patterns["emphasis"]),
378 ]
379
380 while remaining_text:
381 match_found = False
382
383 for pattern_name, pattern in patterns:
384 match = pattern.search(remaining_text)
385 if match:
386 # Add text before match
387 if match.start() > 0:
388 text_before = remaining_text[: match.start()]
389 if text_before:
390 nodes.append(
391 ASTNode(type=NodeType.TEXT, content=text_before)
392 )
393
394 # Add matched element
395 if pattern_name == "image":
396 alt, src = match.groups()
397 nodes.append(
398 ASTNode(
399 type=NodeType.IMAGE, attributes={"alt": alt, "src": src}
400 )
401 )
402 elif pattern_name == "link":
403 text, href = match.groups()
404 link_content = self._parse_inline(text)
405 nodes.append(
406 ASTNode(
407 type=NodeType.LINK,
408 content=link_content,
409 attributes={"href": href},
410 )
411 )
412 elif pattern_name == "inline_code":
413 code = match.group(1)
414 nodes.append(ASTNode(type=NodeType.INLINE_CODE, content=code))
415 elif pattern_name == "inline_math":
416 math = match.group(1)
417 nodes.append(
418 ASTNode(
419 type=NodeType.INLINE_MATH,
420 content=math,
421 attributes={"format": "latex"},
422 )
423 )
424 elif pattern_name == "strong":
425 strong_text = match.group(1)
426 content = self._parse_inline(strong_text)
427 nodes.append(ASTNode(type=NodeType.STRONG, content=content))
428 elif pattern_name == "emphasis":
429 emphasis_text = match.group(1)
430 content = self._parse_inline(emphasis_text)
431 nodes.append(ASTNode(type=NodeType.EMPHASIS, content=content))
432
433 remaining_text = remaining_text[match.end() :]
434 match_found = True
435 break
436
437 if not match_found:
438 # Add remaining text
439 if remaining_text:
440 nodes.append(ASTNode(type=NodeType.TEXT, content=remaining_text))
441 break
442
443 return nodes

449 def _node_to_dict(self, node: ASTNode) -> Dict:
450 """Convert node to dictionary recursively"""
451 result = {"type": node.type.value, "attributes": node.attributes}
452
453 if node.content is not None:
454 if isinstance(node.content, str):
455 result["content"] = node.content
456 elif isinstance(node.content, list):
457 result["content"] = [

461 result["content"] = self._node_to_dict(node.content)

463 if node.position:
464 result["position"] = node.position

504 def extract_links(self, ast: ASTNode) -> List[Dict[str, str]]:
505 """Extract all links from AST"""
506 links = []
507
508 def traverse(node):
509 if node.type == NodeType.LINK:
510 links.append(
511 {
512 "text": self._extract_text_content(node),
513 "href": node.attributes.get("href", ""),
514 }
515 )
516 elif node.type == NodeType.IMAGE:
517 links.append(
518 {
519 "text": node.attributes.get("alt", ""),
520 "href": node.attributes.get("src", ""),
521 }
522 )
523
524 # Recursively process children
525 if isinstance(node.content, list):
526 for child in node.content:
527 traverse(child)
528 elif isinstance(node.content, ASTNode):
529 traverse(node.content)
530
531 traverse(ast)
532 return links

534 def _extract_text_content(self, node: ASTNode) -> str:
535 """Extract plain text from a node"""
536 if node.type == NodeType.TEXT:
537 return node.content or ""
538 elif node.type in [

118 def extract_links(self) -> List[str]:
119 """Extract all links from current page"""
120 links = []
121
122 try:
123 # Find all anchor elements
124 anchor_elements = self.driver.find_elements(By.TAG_NAME, "a")
125
126 for anchor in anchor_elements:
127 try:
128 href = anchor.get_attribute("href")
129 if href and self.is_same_domain(href):

133 # Remove fragments and query parameters for uniqueness

136 ._replace(fragment="", query="")
137 .geturl()
138 )
139
140 if clean_url not in self.visited_urls:
141 links.append(absolute_url)
142
143 except Exception as e:
144 logger.debug(f"Error processing link: {e}")
145 continue
146
147 except Exception as e:
148 logger.error(f"Error extracting links from {self.driver.current_url}: {e}")

249 def html_to_markdowntex(self, html_content: str) -> str:
250 """Convert HTML to MarkdownTeX preserving structure and links"""
251
252 # Basic HTML to Markdown conversion
253 # This is a simplified version - in production, use html2text or similar
254
255 # Extract and preserve inline LaTeX
256 tex_patterns = [
257 (r"\$\$([^$]+)\$\$", r"\\begin{equation}\1\\end{equation}"),
258 (r"\$([^$]+)\$", r"\\(\1\\)"),
259 (
260 r"\\begin\{([^}]+)\}(.*?)\\end\{\1\}",
261 lambda m: f"\\begin{{{m.group(1)}}}{m.group(2)}\\end{{{m.group(1)}}}",
262 ),
263 ]
264
265 # Convert HTML structure
266 markdown = html_content
267
268 # Preserve LaTeX
269 for pattern, replacement in tex_patterns:
270 markdown = re.sub(pattern, replacement, markdown, flags=re.DOTALL)
271
272 # HTML to Markdown basic conversion
273 conversions = [
274 (
275 r"<h([1-6])[^>]*>(.*?)</h[1-6]>",
276 lambda m: f"{'#' * int(m.group(1))} {m.group(2).strip()}\\n\\n",
277 ),
278 (r"<strong[^>]*>(.*?)</strong>", r"**\1**"),
279 (r"<em[^>]*>(.*?)</em>", r"*\1*"),
280 (r"<code[^>]*>(.*?)</code>", r"`\1`"),
281 (r'<a[^>]*href="([^"]*)"[^>]*>(.*?)</a>', r"[\2](\1)"),
282 (r'<img[^>]*src="([^"]*)"[^>]*alt="([^"]*)"[^>]*>', r"![\2](\1)"),
283 (r"<p[^>]*>(.*?)</p>", r"\1\\n\\n"),
284 (r"<br[^>]*>", r"\\n"),
285 (r"<ul[^>]*>(.*?)</ul>", lambda m: self.convert_list(m.group(1), "ul")),
286 (r"<ol[^>]*>(.*?)</ol>", lambda m: self.convert_list(m.group(1), "ol")),
287 (r"<li[^>]*>(.*?)</li>", r"- \1\\n"),
288 ]
289
290 for pattern, replacement in conversions:
291 markdown = re.sub(
292 pattern, replacement, markdown, flags=re.DOTALL | re.IGNORECASE
293 )
294
295 # Clean up extra whitespace and HTML tags
296 markdown = re.sub(r"<[^>]+>", "", markdown)
297 markdown = re.sub(r"\\n\\s*\\n", "\\n\\n", markdown)
298 markdown = markdown.strip() + "\\n"
299
300 return markdown

302 def convert_list(self, list_html: str, list_type: str) -> str:
303 """Convert HTML lists to Markdown"""
304 items = re.findall(r"<li[^>]*>(.*?)</li>", list_html, re.DOTALL | re.IGNORECASE)
305
306 if list_type == "ol":
307 return "\\n".join(
308 f"{i + 1}. {item.strip()}" for i, item in enumerate(items)
309 )
310 else:
311 return "\\n".join(f"- {item.strip()}" for item in items)

392 def recursive_scrape(self, max_pages: int = 1000, max_depth: int = 10) -> None:
393 """Recursively scrape entire website"""
394 if not self.setup_driver():
395 return
396
397 try:
398 # Start with base URL
399 urls_to_scrape = [self.base_url]
400 current_depth = 0
401
402 while (
403 urls_to_scrape
404 and current_depth < max_depth
405 and len(self.downloaded_urls) < max_pages
406 ):
407 logger.info(
408 f"Depth {current_depth}, URLs to scrape: {len(urls_to_scrape)}, Downloaded: {len(self.downloaded_urls)}"
409 )
410
411 new_urls_to_scrape = []
412
413 for url in urls_to_scrape:
414 if (
415 url not in self.visited_urls
416 and len(self.downloaded_urls) < max_pages
417 ):
418 self.visited_urls.add(url)
419 new_links = self.scrape_page(url)
420 new_urls_to_scrape.extend(new_links)
421
422 urls_to_scrape = list(set(new_urls_to_scrape)) # Remove duplicates
423 current_depth += 1
424
425 # Update site AST metadata
426 self.site_ast["metadata"]["total_pages"] = len(self.downloaded_urls)
427 self.site_ast["metadata"]["failed_pages"] = len(self.failed_urls)
428
429 # Save site AST
430 with open(self.output_dir / "site_ast.json", "w", encoding="utf-8") as f:
431 json.dump(self.site_ast, f, indent=2, ensure_ascii=False)
432
433 logger.info(
434 f"Scraping complete. Downloaded {len(self.downloaded_urls)} pages, failed {len(self.failed_urls)}"
435 )
436
437 finally:
438 if self.driver:
439 self.driver.quit()

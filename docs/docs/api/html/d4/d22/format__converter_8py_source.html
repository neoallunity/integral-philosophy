138 output_format: str,
139 **kwargs,

141 """Convert using Pandoc"""
142
143 # Build Pandoc command
144 cmd = ["pandoc", "-f", input_format, "-t", output_format]
145
146 # Add format-specific options
147 if output_format == "latex":
148 cmd.extend(["--pdf-engine=xelatex", "--variable=geometry:margin=1in"])
149 elif output_format == "html":
150 cmd.extend(["--standalone", "--css=style.css"])
151 elif output_format == "typst":
152 cmd.extend(["--standalone"])
153 # Org mode doesn't need special options
154 else:
155 pass # No special options for other formats
156
157 # Add metadata if provided
158 if "metadata" in kwargs:
159 for key, value in kwargs["metadata"].items():
160 cmd.extend(["-M", f"{key}={value}"])
161
162 # Add bibliography if provided
163 if "bibliography" in kwargs:
164 cmd.extend(["--bibliography", kwargs["bibliography"]])
165
166 # Add citation style if provided
167 if "csl" in kwargs:
168 cmd.extend(["--csl", kwargs["csl"]])
169
170 # Add input and output files
171 cmd.extend([str(input_file), "-o", str(output_file)])
172
173 # Run Pandoc
174 try:
175 result = subprocess.run(
176 cmd,
177 capture_output=True,
178 text=True,
179 cwd=str(self.work_dir),
180 timeout=300, # 5 minute timeout
181 )
182
183 if result.returncode == 0:
184 return True
185 else:
186 logger.error(f"Pandoc error: {result.stderr}")
187 return False

191 return False

359 def _calculate_text_similarity(self, text1: str, text2: str) -> float:
360 """Calculate simple text similarity (can be enhanced)"""
361
362 # Remove common markup to compare content
363 def clean_text(text):
364 import re
365
366 # Remove HTML tags
367 text = re.sub(r"<[^>]+>", "", text)
368 # Remove LaTeX commands
369 text = re.sub(r"\\[a-zA-Z]+\{([^}]*)\}", r"\1", text)
370 # Remove extra whitespace
371 text = " ".join(text.split())
372 return text.lower()
373
374 clean1 = clean_text(text1)
375 clean2 = clean_text(text2)
376
377 if len(clean1) == 0 and len(clean2) == 0:
378 return 1.0
379 if len(clean1) == 0 or len(clean2) == 0:
380 return 0.0
381
382 # Simple word overlap similarity
383 words1 = set(clean1.split())
384 words2 = set(clean2.split())
385
386 intersection = words1.intersection(words2)
387 union = words1.union(words2)
388
389 return len(intersection) / len(union) if union else 0.0

417 def _extract_headings(self, file_path: Path) -> List[str]:
418 """Extract headings from file"""
419
420 format_type = self.detect_format(file_path)
421
422 try:
423 with open(file_path, "r", encoding="utf-8") as f:
424 content = f.read()
425
426 headings = []
427
428 if format_type == "markdown":
429 import re
430
431 # Markdown headings (# ## ###)
432 matches = re.findall(r"^(#{1,6})\s+(.+)$", content, re.MULTILINE)
433 headings = [f"#{len(level)} {text}" for level, text in matches]
434
435 elif format_type == "html":
436 import re
437
438 # HTML headings (h1, h2, h3, etc.)
439 matches = re.findall(
440 r"<h([1-6])[^>]*>([^<]+)</h[1-6]>", content, re.IGNORECASE
441 )
442 headings = [f"h{level} {text}" for level, text in matches]
443
444 elif format_type == "latex":
445 import re
446
447 # LaTeX sections
448 matches = re.findall(
449 r"\\(section|subsection|subsubsection)\*?\{([^}]+)\}", content
450 )
451 headings = [f"\\{cmd}{{{text}}}" for cmd, text in matches]
452
453 elif format_type == "org":
454 import re
455
456 # Org-mode headings
457 matches = re.findall(r"^(\*+)\s+(.+)$", content, re.MULTILINE)
458 headings = [f"{level} {text}" for level, text in matches]
459
460 elif format_type == "rst":
461 import re
462
463 # reST headings (underlined)
464 lines = content.split("\n")
465 for i, line in enumerate(lines):
466 if line.strip() and i < len(lines) - 1:
467 next_line = lines[i + 1]
468 if all(c in "=-~^" for c in next_line.strip()) and len(
469 next_line.strip()
470 ) >= len(line.strip()):
471 headings.append(line)
472
473 return headings
474
475 except Exception as e:
476 logger.error(f"Error extracting headings from {file_path}: {e}")
477 return []

479 def create_format_matrix(self, input_file: Path) -> Dict[str, Any]:

7% Статья: автор как глава, заголовок как ненумерованный раздел, запись в TOC и колонтитулы.
8% Использование:
9% \JournalArticle{Фамилия И.О.}{Название статьи}
10\NewDocumentCommand{\JournalArticle}{m m}{%
11 \chapter{#1}
12 \section*{#2}
13 \addcontentsline{toc}{section}{#2}%
14 \markright{#2}%
15}
16
17% Русская аннотация + ключевые слова
18% \RUAbstract{Текст...}{Ключевые слова: ...}
19\NewDocumentCommand{\RUAbstract}{m m}{%
20 \noindent #1
21
22 \vspace{0.3cm}
23 \noindent\textit{#2}
24 \vspace{0.7cm}
25}
26
27% Английская аннотация + keywords
28% \ENAbstract{Text...}{Keywords: ...}
29\NewDocumentCommand{\ENAbstract}{m m}{%
30 \begin{EN}
31 \noindent #1
32
33 \vspace{0.3cm}
34 \noindent\textit{#2}
35 \end{EN}
36 \vspace{0.7cm}
37}

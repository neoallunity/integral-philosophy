13-- | Enhanced error handling for academic publishing
14safeReadFile :: FilePath -> IO (Either String Text.Text)
15safeReadFile path = Exception.catch
16 (Right <$> TextIO.readFile path)
17 (\e -> pure $ Left $ "Failed to read " ++ path ++ ": " ++ show (e :: Exception.SomeException))
18
19-- | Safe file writing with backup
20safeWriteFile :: FilePath -> Text.Text -> IO (Either String ())
21safeWriteFile path content = Exception.catch
22 (do
23 let backupPath = path ++ ".backup"
24 IO.withFile backupPath IO.WriteMode $ \handle -> TextIO.hPutStr handle content
25 TextIO.writeFile path content
26 pure $ Right ()
27 ) (\e -> pure $ Left $ "Failed to write " ++ path ++ ": " ++ show (e :: Exception.SomeException))

160processFile processor path = do

170-- | Academic metadata extraction
171extractMetadata :: Text.Text -> [(Text.Text, Text.Text)]
172extractMetadata content =
173 let lines' = Text.lines content
174 metadataLines = filter (\line ->
175 "\\title" `Text.isPrefixOf` line ||
176 "\\author" `Text.isPrefixOf` line ||
177 "\\date" `Text.isPrefixOf` line) lines'
178 in map parseMetadataLine metadataLines
179 where
180 parseMetadataLine line =
181 let (key, value) = Text.breakOn "{" line
182 cleanValue = Text.drop 1 $ Text.init value
183 in (Text.tail key, cleanValue)
